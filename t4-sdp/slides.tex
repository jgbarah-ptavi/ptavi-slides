\begin{frame}
\frametitle{Introducción}

\begin{itemize}
\item Estándar originariamente desarrollado para Mbone
\item Se utiliza para descripción de sesiones multimedia de tiempo real
\item Tareas: Anuncio, invitación, y otras formas de iniciación
\item Es sólo una descripción, que se usa en el marco de varios
  protocolos: SIP, HTTP, RTSP, SAP, etc.
\item Tipo MIME: application/sdp (a poner en la cabecera SIP/HTTP)
\item Aprobado como RFC 4566 en 2006
\end{itemize}

\begin{flushright}
\url{http://www.ietf.org/rfc/rfc4566.txt}
\end{flushright}
\end{frame}

\begin{frame}
\frametitle{Conceptos SDP}

\begin{itemize}
\item Sesión: Emisor y receptor(es) multimedia, y el flujo de datos que intercambian 
\item Conferencia: Dos o más usuarios que se comunican, junto con el software que usan. \\
  Una conferencia multimedia es un caso de sesión multimedia.
\item Descripción de sesión: Formato para proporcionar la información necesaria para descubrir y participar en una sesión multimedia.
\item Anuncio de sesión: Mecanismo por el que una descripción de sesión se hace llegar a los usuarios proactivamente (sin que lo pidan).
\end{itemize}

\end{frame}

%%------------------------------------------------------------------

\begin{frame}
\frametitle{Principales elementos}

\begin{itemize}
\item Nombre de la sesión y descripción
\item Momentos en que la sesión está activa
\item Medios que componen la sesión (véase siguiente transparencia)
\item Información para recibir esos medios (direcciones, puertos,
  formatos, etc.)
\item Ancho de banda necesario
\item Información de contacto de responsables de la sesión
\end{itemize}

(No todos estos elementos han de estar siempre presentes)


\end{frame}

%%------------------------------------------------------------------

\begin{frame}
\frametitle{Principales elementos: medios}

\begin{itemize}
\item Tipo de medio: audio, vídeo, etc.
\item Protocolo de transporte: RTP, H.320, etc.
\item Formato del medio: vídeo H.261, vídeo MPEG, etc.)
\item Dirección IP de emisor
\item Puerto de transporte para contacto (normalmente, donde se envían
  los datos, pero también el puerto del servidor de RTSP, por ejemplo)
\end{itemize}

(No todos estos elementos han de estar siempre presentes)


\end{frame}

%%------------------------------------------------------------------

\begin{frame}
\frametitle{Formato (principales elementos)}

Descripción:
\begin{itemize}
\item v: Versión del protocolo
\item o: Creador de la sesión
\item s: Descripción de la sesión
\item i: Información sobre la sesión
\item u: URI con descripción más detallada de la sesión
\item e: Dirección de correo de contacto
\item c: Información de conexión (no siempre necesaria)
\item a: Atributos de la sesión (0 ó más)
\end{itemize}

\end{frame}


%%------------------------------------------------------------------

\begin{frame}
\frametitle{Formato (principales elementos, 2)}

Tiempo:

\begin{itemize}
\item t: Tiempo en que la sesión está activa (NTP en decimal)
\item r: Repeticiones (0 ó más)
\end{itemize}

Descripción de medios:
\begin{itemize}
\item m: Descripción de medio y dirección de transporte asociada
\item c: Información de conexión (si es necesaria)
\item b: Ancho de banda
\item a: Atributos del medio (0 ó más)
\end{itemize}

\end{frame}

\begin{frame}[fragile]
\frametitle{Esquema de una descripción SDP}

\begin{verbatim}
        v=  (protocol version)
        o=  (originator and session identifier)
        s=  (session name)
        i=* (session information)
        u=* (URI of description)
        e=* (email address)
        p=* (phone number)
        c=* (connection information -- 
             not required if included in all media)
        b=* (zero or more bandwidth information lines)
        One or more time descriptions 
        z=* (time zone adjustments)
        k=* (encryption key)
        a=* (zero or more session attribute lines)
        Zero or more media descriptions
\end{verbatim}


\footnotesize{(* denota parámetros opcionales)}

\end{frame}

%%------------------------------------------------------------------
\begin{frame}[fragile]
\frametitle{Esquema de una descripción SDP (2)}

Descripción de tiempos (en Network Time Protocol, segundos desde 1900):

\begin{verbatim}
        t=  (time the session is active)
        r=* (zero or more repeat times)
\end{verbatim}

Descripción de medios:

\begin{verbatim}
        m=  (media name and transport address)
        i=* (media title)
        c=* (connection information -- optional if
             included at session level)
        b=* (zero or more bandwidth information lines)
        k=* (encryption key)
        a=* (zero or more media attribute lines)
\end{verbatim}

\footnotesize{(* denota parámetros opcionales)}

\end{frame}

\begin{frame}[fragile]
\frametitle{Ejemplo (de la RFC)}

\begin{verbatim}
v=0
o=mhandley 2890844526 2890842807 IN IP4 126.16.64.4
s=SDP Seminar
i=A Seminar on the session description protocol
u=http://www.cs.ucl.ac.uk/staff/M.Handley/sdp.03.ps
e=mjh@isi.edu (Mark Handley)
c=IN IP4 224.2.17.12/127
t=2873397496 2873404696
a=recvonly
m=audio 49170 RTP/AVP 0
m=video 51372 RTP/AVP 31
m=application 32416 udp wb
a=orient:portrait
\end{verbatim}

\end{frame}
